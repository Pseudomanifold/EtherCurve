CMAKE_MINIMUM_REQUIRED( VERSION 2.6 )
PROJECT( EtherVis )

SET( CMAKE_MODULE_PATH
  ${CMAKE_MODULE_PATH}
  ${CMAKE_SOURCE_DIR}
)

FIND_PACKAGE( PCAP REQUIRED )

IF( CMAKE_COMPILER_IS_GNUCXX )
  SET( CMAKE_CXX_FLAGS "-Werror -Wall -pedantic-errors" )
ENDIF()

ADD_EXECUTABLE(
  ethervis
  EtherVis.cc
)

TARGET_LINK_LIBRARIES(
  ethervis
  ${PCAP_LIBRARIES}
)
